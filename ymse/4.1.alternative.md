static sortByAuthorAlternative(a, b) {
  // 1) Compare by author's last name
  const ln = a.authorLastName.localeCompare(b.authorLastName, "en", { sensitivity: "base" });
  if (ln) return ln;

  // 2) If last names are equal, compare by first name
  const fn = a.authorFirstName.localeCompare(b.authorFirstName, "en", { sensitivity: "base" });
  if (fn) return fn;

  // 3) If author is the same, compare by title
  const ti = a.title.localeCompare(b.title, "en", { sensitivity: "base" });
  if (ti) return ti;

  // 4) Final tie-breaker: year published
  return a.published - b.published;
}
